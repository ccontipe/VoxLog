311    def suggest_output_filename(self, *args):
312        input_path = self.input_file_path.get()
313        if input_path:
314            base_name = os.path.basename(input_path)
315            file_name_without_ext = os.path.splitext(base_name)[0]
316            data_str = datetime.now().strftime("%Y%m%d%H%M%S")
317            ext = os.path.splitext(base_name)[1].lower()
318            if ext in AUDIO_EXTS:
319                prefix = "Transcrição Áudio"
320            elif ext in TEXT_EXTS:
321                prefix = "Transcrição Texto"
322            else:
323                prefix = "Transcrição"
324            suggested_name = f"{prefix} {file_name_without_ext} {data_str}.txt"
325            self.output_file_path.set(suggested_name)
326        else:

356    def start_processing_thread(self):
357        self.stop_transcription_event.clear()
358        self.reset_timer()

360        input_path = self.input_file_path.get()
361        ext = os.path.splitext(input_path)[1].lower()

367        generate_analysis_option = self.analysis_var.get()
368        generate_solution_option = self.solution_var.get()

372        beam_size = self.beam_size_var.get()
373        compute_type = self.compute_type_var.get()

375        if not input_path:
376            messagebox.showwarning("Entrada Ausente", "Por favor, selecione um arquivo de entrada (áudio ou texto).")
377            logger.warning("Tentativa de processamento sem arquivo selecionado.")
378            return
379        if not output_path:
380            messagebox.showwarning("Saída Ausente", "Por favor, defina o nome e local do arquivo de saída.")
381            logger.warning("Tentativa de processamento sem local de saída definido.")
382            return
383
384        if generate_solution_option and not selected_cloud_platform:
385            messagebox.showwarning("Seleção Ausente", "Por favor, selecione uma Plataforma Cloud para gerar a solução.")
386            logger.warning("Gerar Solução marcado, mas plataforma Cloud não selecionada.")
387            return
388
389        self.transcribe_button.config(state=tk.DISABLED)
390        self.cancel_button.config(state=tk.NORMAL)
391
392        # INICIA O CONTADOR DE TEMPO
393        self.start_timer()
394
395        if is_audio:
396            self.update_progress_label("Iniciando transcrição de áudio...")
397        elif is_text:
398            self.update_progress_label("Iniciando leitura de texto...")
399        else:
400            self.update_progress_label("Tipo de arquivo não suportado. Processo abortado.")
401            messagebox.showerror("Tipo Inválido", "O tipo de arquivo selecionado não é suportado.")
402            self.transcribe_button.config(state=tk.NORMAL)
403            self.cancel_button.config(state=tk.DISABLED)
404            self.stop_timer()
405            return
406
407        processing_thread = threading.Thread(
408            target=self._run_all_modules,
409            args=(input_path, ext, is_audio, is_text, model_size, use_gpu, output_path, generate_analysis_option,
410                  generate_solution_option, selected_cloud_platform,
411                  self.update_progress_label, self.update_progress_bar_value, self.stop_transcription_event,
412                  self.clear_input_fields, beam_size, compute_type)
413        )
414        processing_thread.start()
415
416        self.master.after(100, lambda: self.check_processing_thread(processing_thread))
